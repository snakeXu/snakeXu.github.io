网易微专业-jq源码解析-2.1.1:jq整体架构,核心函数功能

一、juqery无new构建实例


1.$就是jquery的别称，$()就是在创建jquery的实例对象

(function(root) {
    var jQuery = function() {
      console.log(‘生成一个实例对象’);
      return new jQuery(); //想让$()就是创建实例对象?wrong！每次调用$()就执行该function,然后就死循环了
    };

    root.$ = root.jQuery = jQuery; //成功使$成为别称^_^
})(this)


·顺便复习一下new操作符调用构造函数会经历的4步：
(1)创建一个新对象
(2)将构造函数的作用域赋给新对象(因此this就指向了这个新对象)
(3)执行构造函数中的代码(为这个新对象添加属性)
(4)返回新对象

·then，我找到了阮一峰ECMAScript6入门里的一段话：
ES5 的继承，实质是先创造子类的实例对象this，然后再将父类的方法添加到this上面（Parent.apply(this)）。ES6 的继承机制完全不同，实质是先将父类实例对象的属性和方法，加到this上面（所以必须先调用super方法），然后再用子类的构造函数修改this。

now，是时候祭出共享原型设计了


代码来了：
(function(root) {
    var jQuery = function() {
      return new jQuery.prototype.init();
    };
    jQuery.prototype = {
        init:function(){
            //
        }
    }
    jQuery.prototype.init.prototype = jQuery.prototype//共享原型对象
    root.$ = root.jQuery = jQuery;
})(this)

这样我们打印console.log($())，就可以看到jquery原型对象的init方法了

这时候的$()实际指向的是jQuery原型上的init方法，而init的原型和jQuery本身的原型共享一个原型对象，这样往jQuery上扩展的方法就可以通过$()获取。

二、extend


(function(root) {
    var jQuery = function() {
      return new jQuery.prototype.init();
    };
    jQuery.fn = jQuery.prototype = {//与上面代码不同之处
        init:function(){
            //
        }
    }
    jQuery.fn.init.prototype = jQuery.fn //与上面代码不同之处
    root.$ = root.jQuery = jQuery;
})(this)

从而可以使$.fn.extend和$.extend达到同样效果，于是我们开始使用
//jQuery本身扩展

$.extend({
work:function(){}
})

或者

//jQuery实例对象扩展,此时this === jQuery实例对象

$.fn.extend({
sex: ‘男’
})

或者

$.extend({}, { name: ‘arbor’});

对任意对象进行扩展,于是我们可以使用
jQuery.work();
$().sex;

那么，源码是怎么实现这个功能的呢？先来实现浅拷贝


(function(root) {
    var jQuery = function() {
      return new jQuery.prototype.init();
    };
    jQuery.fn = jQuery.prototype = {
        init:function(){
            //
        }
    }
    jQuery.fn.extend = jQuery.extend = function(){
        var target = arguments[0]||{};//arguments[0]一定是一个object
        var length = arguments.length;
        var i=1;
        var option, name;
        if(typeof arguments[0] !== ‘object’){
            target = {}
        }
        //参数个数
        if(length === i){
            target = this;
        }
        //浅拷贝
        for (; i < length; i++) {
            if((option = arguments[i]) != null){
                for(name in option){
                    target[name] = option[name]
                }
            }
        }
        return target;
    }
    jQuery.fn.init.prototype = jQuery.fn
    root.$ = root.jQuery = jQuery;
})(this)

jQuery.fn.extend = jQuery.extend = function() {
    var target = arguments[0] || {};
    var length = arguments.length;
    var i = 1;
    var deep = false;
    var option, name,copy,src,copyIsArray,clone;
    if (typeof target === "boolean") {
        deep = target;
        target = arguments[1];
        i = 2;
    }
    if (typeof target !== "object") {
        target = {};
    }
    //参数的个数 1
    if (length === i) {
        target = this;
        i--;
    }

    //浅拷贝  深拷贝
    for (; i < length; i++) {
        if ((option = arguments[i]) != null) {
            for (name in option) {
                copy = option[name];
                src = target[name];
                if(deep && (jQuery.isPlainObject(copy) || (copyIsArray = jQuery.isArray(copy)))){
                    if(copyIsArray){
                        copyIsArray = false;
                        clone = src && jQuery.isArray(src) ? src : [];
                    } else {
                        clone = src && jQuery.isPlainObject(src) ? src : {};
                    }
                    target[name] = jQuery.extend(deep, clone, copy);//灵魂
                } else if(copy != undefined){
                    target[name] = copy;
                }
            }
        }
    }
    return target;
}
