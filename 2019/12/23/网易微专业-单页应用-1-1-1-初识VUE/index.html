< script src=”https://cdn.bootcss.com/vue/2.6.10/vue.min.js">< /script>
< div id=”app”>
    < ul>
        < li v-for=”item in feBooks”>{ {item.name} }< /li>
    < /ul>
    < input type=”text” v-model=”title”>
    < p>{ {title} }< /p>
< /div>

        el:”#app”,
