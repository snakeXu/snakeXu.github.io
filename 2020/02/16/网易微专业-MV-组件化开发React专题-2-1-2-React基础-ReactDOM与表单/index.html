1.ReactDOM.render()&ReactDOM.unmountComponentAtNode()
ReactDOM.render():简单的说，就是挂载
ReactDOM.unmountComponentAtNode():简单的说就是卸载
2.例子
App.js


import React from ‘react’;
import ReactDOM from ‘react-dom’;
import Loading from ‘./Loading’
class App extends React.Component{
  constructor(props){
    super(props)
    this.state = {
      data:[],
      loading:false,
    }
  }
  render(){
    return (
      < div>
        App
        {this.state.data.map((item)=>< p key={item.id}>{item.name}< /p>)}
        {this.state.loading && < Loading>< /Loading>}
      < /div>
    )
  }
  componentDidMount(){
    //loading show
    this.setState({
      loading:true
    })
    setTimeout(()=>{
      this.setState({
        data:[{name:’a’,id:’a’},{name:’b’,id:’b’},{name:’c’,id:’c’}],
        loading:false
      })
    },3000)
    //loading hide
  }
}
export default App;

Input.js


import React from ‘react’;
import ReactDOM from ‘react-dom’;
let node = null;
class Input extends React.Component{
  constructor(props){
    super(props)
    this.state = {
      value:’’
    }
  }
  render(){
    const {onOk} = this.props
    return (
      < div>
        < input value={this.state.value} onChange={(e)=>this.setState({value:e.target.value})}/>
        < button onClick={(e)=>{
          onOk(this.state.value)
          this.setState({value:’’})
          hide()
        }}>确定< /button>
      < /div>
    )
  }
  componentDidMount(){
  }
}
export const show = (obj)=>{
  node = document.createElement(‘div’)
  document.body.appendChild(node)
  ReactDOM.render(< Input onOk={obj.onOk}>< /Input>,node)
}
export const hide = ()=>{
  if(node){
    ReactDOM.unmountComponentAtNode(node)
    document.body.removeChild(node)
  }
}
export default {
  show,
  hide
}

import React from ‘react’;
import ReactDOM from ‘react-dom’;
import Todolist from ‘./Todolist’
import Action from ‘./Action’
class App extends React.Component{
  constructor(props){
    super(props)
    this.state = {
      data:[]
    }
  }
  render(){
    return (
      < div>
        < Action onAdd={(name)=>{
          let data = this.state.data
          data.push(name)
          this.setState({data})
        }}/>
        < Todolist data={this.state.data}
        onDelete={(index)=>{
          let data = this.state.data
          data.splice(index,1)
          this.setState({data})
        }}
        />
      < /div>
    )
  }
  componentDidMount(){
  }
}
export default App;
