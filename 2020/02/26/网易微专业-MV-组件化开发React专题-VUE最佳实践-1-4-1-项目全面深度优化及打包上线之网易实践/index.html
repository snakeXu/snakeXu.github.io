1-3.工程优化-vue-cli-plugin的实现
在项目目录下新建一个文件夹(假设我们当前的项目目录名为my-app)，取名vue-cli-plugin-demo，新建文件index.js


const CSSSpritePlugin = require(‘css-sprite-loader’).Plugin;
function chainCSSOneOfs(config, chainOneOf) {
    // ‘postcss’ 留给用户自定义吧
    [‘css’, ‘scss’, ‘sass’, ‘less’, ‘stylus’].forEach((rule) => {
        const ruleCSS = config.module.rule(rule);
        chainOneOf(ruleCSS.oneOf(‘vue-modules’), true);
        chainOneOf(ruleCSS.oneOf(‘vue’), false);
        chainOneOf(ruleCSS.oneOf(‘normal-modules’), true);
        chainOneOf(ruleCSS.oneOf(‘normal’), false);
    });
}
module.exports = function (api)  {
    api.chainWebpack(config => {
        config.plugin(‘css-sprite’).use(CSSSpritePlugin, [])
        chainCSSOneOfs(config, (oneOf) => {
            oneOf.use(‘css-sprite-loader’)
                .loader(‘css-sprite-loader’)
                .end();
        });
    })
}

这个文件的可以取代vue.config.js，在vue-cli-plugin-demo文件夹下建一个package.json文件，和平时package.json文件内容一样，在我们新建的package.json的devDependencies加入一项vue-cli-plugin-demo:’^0.1.0’，其实可以用npm publish发布包(npm publish的具体用法可以用这个命令为关键字进行百度)，这里我们为了方便调试包，使用另外一种方法
然后我们在命令行进入当前my-app项目的node_modules文件夹，输入命令ln -s ../vue-cli-plugin-demo，在输入ls命令我们看到vue-cli-plugin-demo与其他目录名称颜色不同，这是一个软链。通过命令ls -al vue-cli-plugin-demo可以看到指向上一级目录，这个命令和npm unlink功能差不多但是用软链的效果快一些，返回项目根目录，npm run serve启动项目，这个时候项目应该是没有报错的
2.部署优化
很多同学会遇到一个问题就是随着项目变大，项目打包速度越来越慢
首先我们了解一下发包的过程:如果想发包，要有一个npm账号，在npmjs.com网站注册一个账号，然后在命令行输入npm login登陆，进入需要发包的目录(目录中需要有package.json文件)，在命令行输入npm publish
2-1.部署流程


通过课程我们了解到可以通过jenkins(https://jenkins.io/zh/)/docker写脚本完成打包部署等工作


const fs = require(‘fs’);
const path = require(‘path’);
const shell = require(‘shelljs’);
const crypto = require(‘crypto’);
const md5 = function (str, len = 16) {
    const md5 = crypto.createHash(‘md5’);
    return md5.update(str).digest(‘hex’).substr(0, len);
};
const cachePath = path.join(dirname, ‘./node_modules/.cache/package-lock.log’);
if (!fs.existsSync(path.join(cachePath, ‘../..’))) {
    fs.mkdirSync(path.join(cachePath, ‘../..’), {
        recursive: true,
    });
}
const lockHash = md5(fs.readFileSync(path.join(dirname, ‘./package-lock.json’)).toString());
let cache = false;
if (fs.existsSync(cachePath)) {
    if (lockHash === fs.readFileSync(cachePath).toString()) {
        cache = true;
        console.log(‘Installed from cache.’);
    }
}
if (!cache) {
    /**

 * Install npm packages
 */
console.time('npm install');
console.log('Installing npm packages ...');
const result = shell.exec('npm --no-audit --progress=false --registry=https://registry.npm.taobao.org install');
if (result.code)
    throw new Error(String(result.stderr));
/**
 * Log Versions
 */
const nodeVersion = shell.exec('node -v', { silent: true }).stdout.trim();
console.log('node version:', nodeVersion);
const npmVersion = shell.exec('npm -v', { silent: true }).stdout.trim();
console.log('npm version:', npmVersion);
console.timeEnd('npm install');
fs.writeFileSync(cachePath, lockHash);

}

上面代码我们可以起名为npm.js，放在项目的根目录下，比如我们的my-app项目的根目录下。每次安装项目都会按照package-lock.json重新安装项目依赖，导致资源浪费，有了npm.js，我们执行node npm命令，此后每次安装项目，如果package.json文件没有变化，不会重新安装。
2-2.webpack.dll优化
这里主要利用webpack两个相关的plugin，一个是DllPlugin，另一个是DllReferencePlugin。我们可以在项目中配webpack.dll.config.js文件，然后通过配置把不经常需要更换的第三方依赖/库等打包，然后在vue.config.js中通过DllReferencePlugin引入，另外还需要一个AddAssetHtmlPlugin把我们通过dll打包的文件加入到html页面
用这个优化会比用commonschunks或者splitchunks某种程度上更节省时间，因为用dll优化不会每次都打包指定内容，但是用commonschunk或者是splitchunk其实每次还是打包的，所以对于一些极少更新的依赖，比如babel-polyfill等还是用dll优化
另外可以看一些别的打包优化方式，比如提取公共库等
3.组件优化
3-1.虚拟列表
vue文件，这里只写了页面代码，项目可以自己创建一个


< template>
< div :class=”$style.list” ref=”list” @scroll=”onScroll”>
    < div :class=”$style.scroll” :style=”{ paddingTop: paddingTop + ‘px’, paddingBottom: paddingBottom + ‘px’ }”>
        < div :class=”$style.item” v-for=”item in viewItems” :key=”item.id” :even=”!!(item.id%2)”>
            < div v-for=”cnt in item.content” :key=”cnt”>{ { cnt } }< /div>
        < /div>
    < /div>
< /div>
< /template>
< script>
export default {
    props: {
        list: Array,
        viewCount: { type: Number, default: 100 },
    },
    data() {
        return {
            minHeight: 20,
            viewIndex: 0,
            paddingTop: 0,
            paddingBottom: 0,
        }
    },
    computed: {
        viewItems() {
            return this.list.slice(this.viewIndex, this.viewIndex + this.viewCount);
        },
    },
    methods: {
        onScroll(e) {
            const listEl = e.target;
            // 记录当前 DOM 节点的高度
            const children = Array.from(listEl.children[0].children);
            children.forEach((childVM, index) => {
                if (this.list[this.viewIndex + index])
                    this.list[this.viewIndex + index].height = childVM.offsetHeight;
            });
            / eslint-disable /
            const scrollTop = listEl.scrollTop;
            let accHeight = 0;
            let viewIndex = this.viewIndex;
            for (let i = 0; i < this.list.length; i++) {
                const item = this.list[i];
                accHeight += item.height || this.minHeight;
                if (accHeight > scrollTop) {
                    viewIndex = i;
                    break;
                }
            }
            if (viewIndex - this.viewIndex >= 0 && viewIndex - this.viewIndex < this.viewCount / 2)
                return;
            let paddingTop = 0;
            let paddingBottom = 0;
            for (let i = 0; i < this.list.length; i++) {
                const item = this.list[i];
                if (i < viewIndex) {
                    paddingTop += item.height || this.minHeight;
                } else if (i >= viewIndex + this.viewCount) {
                    paddingBottom += item.height || this.minHeight;
                }
            }
            this.viewIndex = viewIndex;
            this.paddingTop = paddingTop;
            this.paddingBottom = paddingBottom;
        },
    },
}
< /script>
< style module>
.list {
    overflow: auto;
    width: 100%;
    height: 100%;
}
.item[even] {
    background: #c3c9d3;
}
< /style>
