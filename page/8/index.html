import React,{ Component } from ‘react’
import PropTypes from ‘prop-types’
class Router extends Component{
    render(){
        return (
            < >
            {this.props.children}
            </>
        )
    }
}
class Route extends Component{
    constructor(props){
        super(props);
        this.state = {
            hash:window.location.hash
        }
    }
    static contextTypes = {
        hash: PropTypes.string
    }
    getHash(){
        let url = window.location.hash.replace(‘#’,’’)
        return url
    }
    getChildContext(){
        return {
            hash:this.getHash()
        }
    }
    componentDidMount(){
        window.onhashchange = ()=>{
            this.setState({
                hash:this.getHash()
            })
        }
    }
    render(){
        const {
            path,
            component
        } = this.props;
        let instance = null;
        //let hash = window.location.hash.replace(‘#’,’’)
        const {
            hash
        } = this.context
        if(path === hash){
            //React.createElement(component,{name:’a’,age:2},children)，即第二个可以传组件需要参数，第三个参数是子组件
            instance = React.createElement(component,null,null);
        }
        return (
            < >
            {instance}
            </>
        )
    }
}

import React,{ Component } from ‘react’
import PropTypes from ‘prop-types’
class Router extends Component{
    constructor(props){
        super(props);
        this.state = {
            hash:this.getHash()
        }
        this.history = window.history
    }
    static childContextTypes = {
        hash: PropTypes.string,
    }
    getHash(){
        let url = window.location.pathname;
        return url
    }
    getChildContext(){
        return {
            hash:this.state.hash
        }
    }
    componentDidMount(){
        this.history.route = (name) =>{
            this.setState({
                hash:${name}
            })
            window.history.pushState(null,null,name);//只是改变页面状态，没有做任何请求
        }
    }
    render(){
        return (
            < >
            {this.props.children}
            </>
        )
    }
}

    static contextTypes = {
        hash: PropTypes.string
    }
    render(){
        const {
            path,
            component

        } =this.context

            //React.createElement(component,{name:’a’,age:2},children)，即第二个可以传组件需要参数，第三个参数是子组件
            instance = React.createElement(component,null,null);
        }
        return (
            < >
            {instance}
            </>
        )
    }
}

4-1.如何区分前后端路由
如果url中既包含前端路由又包含后端路由，我们通常看到node端路由会有app.use(express为例)等指定路由，那么app.use中指定的就是后端路由

2.代码分割
2-1.多页面应用
提取公共依赖：把几个页面之中都用到的依赖给打包为一个单独的文件
2-2.单页面应用

2-3.为了业务代码的纯净

2-4.所以一般这么打包

网易微专业-webpack实战演练-3.2.3-图片等资源的处理

1.需要用到哪些loader
file-loader:webpack识别除js之外的资源(不用这个loader的时候也是识别js资源的)
url-loader:url-loader包含了file-loader
img-loader:对图片进行压缩处理等需要这个loader

在没有安装这些loader的时候，webpack打包不能正确识别图片资源，示例代码初步了解

常见的坑: (1).路径问题： 
{
    test:/\.(jpg|jpeg|gif|png)$/,

        options:{

        }
    }
}
如果我们按照上面的设置了publicPath，那么打包出来的文件路径就变成了background:url(aaa/img.39e0.jpg)，一般会把outputPath和publicPath设置成一样的 (2).命名问题： 1-2.url-loader 在file-loader的基础上增加了可以将资源转换为base64的功能，默认可以将小于30kb的资源转换为base64 
{
    test:/\.(jpg|jpeg|gif|png)$/,
    use:{
        loader:'url-loader',
        options:{
            name:'[name].[hash:4].[ext]',
            outputPath:'assets/img',
            publicPath:'aaa',
            //我们为什么不把所有东西都转换为base64？
            //虽然我们将资源转换为base64减少了我们的请求，但是资源转换为base64的字符串和资源本身的大小是一样的，
            //比如我们需要的图片是5kb，那么转换为base64之后这5kb会加载到css文件中，css文件变大造成该文件加载变慢
            limit:5000,//控制将多大的资源转换为base64，一般设置为5kb
        }
    }
}
1-3.img-loader 可以压缩图片。loader是从后往前执行的，img-loader只是处理图片，所以我们把img-loader放在url-loader下面。这里要明白img-loader本身不做任何事情，所以并不是写上img-loader就完事儿的，需要配置一系列插件 
{
    test:/\.(jpg|jpeg|gif|png)$/,
    use:[
        {
            loader:'url-loader',
            options:{
                name:'[name].[hash:4].[ext]',
                outputPath:'assets/img',
                publicPath:'aaa',
                limit:5000,
            }
        },
        //压缩图片的插件imagemin系列，对于不同的图片类型有不同的包
        //处理png图片：imagemin-pngquant 处理jpeg图片：imagemin-mozjpeg 处理gif图片：imagemin-gifsicle
        //imagemin系列包都是要通过npm下载安装的
        {
            loader:'img-loader',
            options:{
                plugins:[
                    require('imagemin-pngquant')({
                        speed:2,//控制压缩质量，值越大，对图片压缩程度越小，取值范围为1~11
                    }),
                    require('imagemin-mozjpeg')({
                        quality:2,//控制压缩质量，值越大，对图片压缩程度越小，取值范围为1~100
                    }),
                    require('imagemin-gifsicle')({
                        optimizationLevel:2,//取值范围为1,2,3，推荐设置为1，设置为3极有可能失真
                    }),
                ]
            }
        }
    ]
}

const routerLock = {
    beforeEach(to, from, next) {
        const matched = to.matched || [];
        const locks = [];
        matched.forEach((item) => {
            if (item.meta && item.meta.locks) {
                const itemLocks = item.meta.locks;
                itemLocks.forEach((lock) => {
                    if (lock.params && lock.params.length) {
                        locks.push(lock);
                    }
                });
            }
        });
        if (!locks.length) {
            next();
            return;
        }
        const includePath = function (path) {
            if (typeof path === ‘string’) {
                return path === to.name || path === to.path;
            }
            if (path instanceof RegExp) {
                return path.test(to.name) || path.test(to.path);
            }
            return false;
        };
        const fromQuery = from.query;
        const toQuery = to.query;
        let isChanged = false;
        locks.forEach((lock) => {
            // 没有 include 时全通过
            if (lock.include && !lock.include.some(includePath))
                return;
            if (lock.exclude && lock.exclude.some(includePath))
                return;
            lock.params.forEach((param) => {
                if (!(param in toQuery)) {
                    isChanged = true;
                    toQuery[param] = fromQuery[param];
                }
                if (toQuery[param] === ‘’) {
                    toQuery[param] = undefined;
                } // 美化路由
            });
        });
        if (isChanged) {
            next({
                path: to.path,
                query: toQuery,
            });
        } else {
            next();
        }
    },
    update(query, route, router) {
        if (Object.keys(query).some((key) => query[key] !== route.query[key])) {
            router.push({
                path: route.path,
                query: {
                    …route.query,
                    …query,
                },
            });
        }
    },
    install(Vue) {
        Vue.prototype.$routerLock = function (query) {
            return routerLock.update(query, this.$route, this.$router);
        };
    },
};
export default routerLock;
